	\begin{eqnarray*}
		\Delta_1 q_1(\bm{x})
		&=& a_{11} q(\bm{x})\\

        &=&
a_{11}^2x_1^2+2a_{11}a_{12}x_1x_2+a_{11}a_{22}x^2\\
&=&
a_{11}^2x_1^2+2a_{11}a_{12}x_1x_2+a_{12}^2x_2-a_{12}^2x_2+a_{11}a_{22}x^2\\
&=&(a_{11}x_1+a_{12}x_2)^2+(a_{11}a_{22}-a_{12}^2)x_2^2\\
&=&(a_{11}x_1+a_{12}x_2)^2+
\begin{vmatrix}
a_{11} & a_{12}\\
a_{12} & a_{22}
\end{vmatrix}x_2^2\\
&=&(a_{11}x_1+a_{12}x_2)^2+
\Delta_2x^2
\end{eqnarray*}
\item[正：] 
	\begin{eqnarray*}
		\Delta_1 q_1(\bm{x})
		&=& a_{11} q(\bm{x})\\
		&=& a_{11}(a_{11}x_1^2+2a_{12}x_1x_2+a_{22}x^2)\\
        &=&
a_{11}^2x_1^2+2a_{11}a_{12}x_1x_2+a_{11}a_{22}x^2\\
&=&
a_{11}^2x_1^2+2a_{11}a_{12}x_1x_2+a_{12}^2x_2^{\textcolor{red}{2}}-a_{12}^2x_2^{\textcolor{red}{2}}+a_{11}a_{22}x^2\\
&=&(a_{11}x_1+a_{12}x_2)^2+(a_{11}a_{22}-a_{12}^2)x_2^2\\
&=&(a_{11}x_1+a_{12}x_2)^2+
\begin{vmatrix}
a_{11} & a_{12}\\
a_{12} & a_{22}
\end{vmatrix}x_2^2\\
&=&(a_{11}x_1+a_{12}x_2)^2+
\Delta_2x^2_{\textcolor{red}{2}}
\end{eqnarray*}
